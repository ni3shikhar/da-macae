## 8. SQL Scripts — Watermark Infrastructure

### staging-ddl.sql
```sql
-- ═══════════════════════════════════════════════════
-- DA-MACAÉ: ADF watermark infrastructure
-- ═══════════════════════════════════════════════════

IF NOT EXISTS (SELECT 1 FROM sys.tables WHERE name = 'WatermarkTable')
BEGIN
    CREATE TABLE dbo.WatermarkTable (
        TableName       NVARCHAR(256) NOT NULL PRIMARY KEY,
        WatermarkValue  DATETIME2     NOT NULL DEFAULT '1900-01-01'
    );
END;
GO

-- Seed watermark rows
INSERT INTO dbo.WatermarkTable (TableName, WatermarkValue)
SELECT t.TableName, '1900-01-01'
FROM (VALUES
    ('dbo.Customers'),
    ('dbo.Orders'),
    ('dbo.OrderItems'),
    ('dbo.Products'),
    ('dbo.Employees')
) AS t(TableName)
WHERE NOT EXISTS (
    SELECT 1 FROM dbo.WatermarkTable w WHERE w.TableName = t.TableName
);
GO

-- Update watermark stored procedure
CREATE OR ALTER PROCEDURE dbo.usp_UpdateWatermark
    @tableName      NVARCHAR(256),
    @watermarkValue DATETIME2
AS
BEGIN
    SET NOCOUNT ON;
    UPDATE dbo.WatermarkTable
    SET WatermarkValue = @watermarkValue
    WHERE TableName = @tableName;
END;
GO
```
